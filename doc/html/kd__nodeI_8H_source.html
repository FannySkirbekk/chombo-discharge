170 template <typename T> void kd_node<T>::split_data(){
171  CH_assert(m_leaf);
172  CH_assert(!m_left.isNull());
173  CH_assert(!m_right.isNull());
174 
175  Vector<RefCountedPtr<T> > data_right;
176  Vector<RefCountedPtr<T> > data_left;
177 
178  // Find maximum and minimum coordinates (i.e. the bounding box of the data centroids)
179  Vector<RealVect> coords;
180  for (int i = 0; i < m_data.size(); i++){
181  coords.push_back(m_data[i]->get_coord());
182  }
183 
184  RealVect min_coord = 1.234567E89*RealVect::Unit;
185  RealVect max_coord = -1.234567E89*RealVect::Unit;
186  for (int i = 0; i < m_data.size(); i++){
187  for (int dir = 0; dir < SpaceDim; dir++){
188  if(coords[i][dir] < min_coord[dir]){
189  min_coord[dir] = coords[i][dir];
190  }
191  if(coords[i][dir] > max_coord[dir]){
192  max_coord[dir] = coords[i][dir];
193  }
194  }
195  }
196 
197  // Find the longest dimension
198  Real longest_len = 0.0;
199  int longest_dir = 0;
200  for (int dir = 0; dir < SpaceDim; dir++){
201  const Real cur_longest = max_coord[dir] - min_coord[dir];
202  CH_assert(cur_longest >= 0.0);
203  if(cur_longest > longest_len){
204  longest_len = cur_longest;
205  longest_dir = dir;
206  }
207  }
208 
209  // Split data on a left/right structure
210  const RealVect split_pos = min_coord + 0.5*longest_len*RealVect(BASISV(longest_dir));
211  for (int i = 0; i < m_data.size(); i++){
212  if(m_data[i]->get_coord()[longest_dir] < split_pos[longest_dir]){
213  data_left.push_back(m_data[i]);
214  }
215  else{
216  data_right.push_back(m_data[i]);
217  }
218  }
219 
220  if(!m_left.isNull()){
221  m_left->set_data(data_left);
222  m_left->compute_bbox();
223  }
224  else{

226  }
